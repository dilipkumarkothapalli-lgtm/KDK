            { type: 'piped', url: 'https://pipedapi.kavin.rocks' },
            { type: 'invidious', url: 'https://invidious.snopyta.org' },
            { type: 'invidious', url: 'https://invidious.kavin.rocks' },
            { type: 'invidious', url: 'https://yewtu.be' }
        ];

        const fetchWithTimeout = async (resource, options = {}, timeout = 3000) => {
            const controller = new AbortController();
            const id = setTimeout(() => controller.abort(), timeout);
            const response = await fetch(resource, { ...options, signal: controller.signal });
            clearTimeout(id);
            return response;
        };

        async function getVideoId(query) {
            const encodedQuery = encodeURIComponent(query);
            for (const api of API_INSTANCES) {
                try {
                    let apiUrl = '', parser = null;
                    if (api.type === 'piped') {
                        apiUrl = `${api.url}/search?q=${encodedQuery}&filter=music_songs`;
                        parser = (data) => {
                            if (data.items && data.items.length > 0) {
                                const match = data.items[0].url.match(/v=([^&]+)/);
                                return match ? match[1] : null;
                            }
                            return null;
                        };
                    } else if (api.type === 'invidious') {
                        apiUrl = `${api.url}/api/v1/search?q=${encodedQuery}`;
                        parser = (data) => {
                            if (Array.isArray(data) && data.length > 0) return data[0].videoId;
                            return null;
                        };
                    }
                    if (!parser) continue;
                    const response = await fetchWithTimeout(apiUrl);
                    if (!response.ok) continue;
                    const data = await response.json();
                    const videoId = parser(data);
                    if (videoId) return videoId;
                } catch (error) { continue; }
            }
            return null;
        }

        function getBestImage(images) {
            if (!images || images.length === 0) return 'https://picsum.photos/seed/music/300/300';
            const img = images.find(i => i.quality === '500x500');
            return img ? img.url : images[images.length - 1].url;
        }

        function getBestAudio(downloadUrls) {
            if (!downloadUrls || downloadUrls.length === 0) return { url: null, quality: 'Unknown' };
            const flac = downloadUrls.find(x => x.quality === 'flac' || x.quality === 'lossless');
            if (flac) return { url: flac.url, quality: 'FLAC' };
            const hq = downloadUrls.find(x => x.quality === '320kbps');
            if (hq) return { url: hq.url, quality: '320kbps' };
            const mq = downloadUrls.find(x => x.quality === '160kbps');
            if (mq) return { url: mq.url, quality: '160kbps' };
            const lq = downloadUrls.find(x => x.quality === '96kbps');
            if (lq) return { url: lq.url, quality: '96kbps' };
            return { url: downloadUrls[0].url, quality: 'Standard' };
        }

        // --- DATA FETCHING ---
        async function fetchSongs(query, limit = 10, useLanguage = true) {
            try {
                let finalQuery = query;
                if (useLanguage && currentLanguage !== 'all') finalQuery = `${query} ${currentLanguage}`;
                const url = `${API_BASE}/search/songs?query=${encodeURIComponent(finalQuery)}&limit=${limit}`;
                const res = await fetch(url);
                if (!res.ok) throw new Error("API Error");
                const json = await res.json();
                if (json.success && json.data && json.data.results) {
                    return json.data.results.map(track => {
                        const audioData = getBestAudio(track.downloadUrl);
                        return {
                            id: track.id, title: track.name,
                            artist: track.artists?.primary?.[0]?.name || 'Unknown Artist',
                            image: getBestImage(track.image),
                            url: audioData.url, quality: audioData.quality,
                            album: track.album?.name || '', year: track.year || ''
                        };
                    });
                }
            } catch (e) {}
            return [];
        }

        async function fetchAlbums(query, limit = 5, useLanguage = true) {
            try {
                let finalQuery = query;
                if (useLanguage && currentLanguage !== 'all') finalQuery = `${query} ${currentLanguage}`;
                const res = await fetch(`${API_BASE}/search/albums?query=${encodeURIComponent(finalQuery)}&limit=${limit}`);
                const json = await res.json();
                if (json.success && json.data && json.data.results) {
                    return json.data.results.map(album => ({
                        id: album.id, title: album.name,
                        artist: album.artists?.primary?.[0]?.name || 'Various Artists',
                        image: getBestImage(album.image),
                        type: 'album', songCount: album.songCount || 0
                    }));
                }
            } catch (e) { }
            return [];
        }

            if (!items || items.length === 0) { container.innerHTML = '<div style="color:#666; font-size:0.9rem; padding:10px;">No results.</div>'; return; }

                card.tabIndex = 0;

                    ? `<button class="playlist-remove-btn focusable focusable-round" tabindex="-1" onclick="removeFromMyPlaylist('${item.id}', event)"><i class="ri-close-line"></i></button>`
                    : `<button class="card-menu-btn focusable focusable-round" tabindex="-1" onclick="openContextMenu(event, '${type}')"><i class="ri-more-2-fill"></i></button>`;

                card.innerHTML = `
                    <div class="card-img-wrapper">
                        <img src="${item.image}" class="card-img" loading="lazy" alt="${item.title}">
                        <img src="https://i.ibb.co/6R4pnJmB/image.png" class="card-kdk-badge" alt="KDK Badge">

            });
        }

                const json = await res.json();

                    const albumData = json.data;
                    let songs = albumData.songs.map(s => {
                        const audioData = getBestAudio(s.downloadUrl);
                        return {
                            id: s.id, title: s.name,
                            artist: s.artists?.primary?.[0]?.name || albumData.artists?.primary?.[0]?.name,
                            image: getBestImage(s.image),
                            url: audioData.url, quality: audioData.quality,
                            album: albumData.name, year: albumData.year
                        };
                    });
                    if (songs.length === 0) { showToast("Album empty."); return; }
                    mainContent.innerHTML = `
                        <div class="section-header"><button class="section-link focusable" tabindex="0" onclick="loadHome()" style="display:flex; align-items:center; gap:5px; background:none; border:none;"><i class="ri-arrow-go-back-line"></i><span>Back</span></button></div>

        function openContextMenu(e, type) {
            e.stopPropagation();
            const card = e.target.closest('.card');
            const item = JSON.parse(card.dataset.item);
            const list = JSON.parse(card.dataset.list);
            contextMenuTargetSong = item;
            contextMenuSourceList = list;
            const x = e.clientX, y = e.clientY, menuWidth = 200, menuHeight = 130, winWidth = window.innerWidth, winHeight = window.innerHeight;
            let posX = x, posY = y;
            if (x + menuWidth > winWidth) posX = x - menuWidth;
            if (y + menuHeight > winHeight) posY = y - menuHeight;
            contextMenu.style.left = `${posX}px`; contextMenu.style.top = `${posY}px`;
            contextMenu.classList.add('active');
            // Auto focus first item for remote
            setTimeout(()=> { contextMenu.querySelector('.context-item').focus(); }, 50);
        }

        function contextAction(action) {
            if (!contextMenuTargetSong) return;
            if (action === 'playNow') playSong(contextMenuTargetSong, contextMenuSourceList);
            else if (action === 'playNext') {
                if (currentPlaylist.length === 0) playSong(contextMenuTargetSong, [contextMenuTargetSong]);
                else { currentPlaylist.splice(currentIndex + 1, 0, contextMenuTargetSong); renderQueue(); showToast("Added to Play Next"); }
            } else if (action === 'addToQueue') {
                if (currentPlaylist.length === 0) playSong(contextMenuTargetSong, [contextMenuTargetSong]);
                else { currentPlaylist.push(contextMenuTargetSong); renderQueue(); showToast("Added to Queue"); }
            }
            contextMenu.classList.remove('active');
        }

            currentSongObj = song;

            updatePlayerUI(song);

                audio.src = song.url;

                    isPlaying = true; 
                    updatePlayBtns(); 
                    renderQueue(); 
                    fetchLyrics(song);

        }

        function setupSwipeGesture() {
            let touchStartY = 0, touchEndY = 0;
            fpQueueSection.addEventListener('touchstart', e => { touchStartY = e.changedTouches[0].screenY; }, false);
            fpQueueSection.addEventListener('touchend', e => { touchEndY = e.changedTouches[0].screenY; handleSwipe(); }, false);
            function handleSwipe() {
                const diff = touchStartY - touchEndY, threshold = 50;
                if (diff > threshold) { fpQueueSection.classList.add('expanded'); }
                else if (diff < -threshold) { fpQueueSection.classList.remove('expanded'); }
            }
        }

        function parseLRC(lrc) {
            const lines = lrc.split('\n'), result = [], timeReg = /\[(\d{2}):(\d{2})\.(\d{2,3})\]/;
            for (const line of lines) {
                const match = timeReg.exec(line);
                if (match) {
                    const min = parseInt(match[1]), sec = parseInt(match[2]), ms = parseFloat("0." + match[3]), time = min * 60 + sec + ms, text = line.replace(timeReg, '').trim();
                    if (text) result.push({ time, text });
                }
            }
            return result;
        }

        async function translateLyrics(targetLang) {
            if (!originalPlainLyrics) {
                if (lyricsData.length > 0) originalPlainLyrics = lyricsData.map(l => l.text).join('\n');
                else { lyricsContent.innerHTML = '<p style="color:#666; margin-top:50px;">No lyrics.</p>'; return; }
            }
            lyricsContent.innerHTML = '<div class="spinner"></div>';
            try {
                const verses = originalPlainLyrics.split('\n\n'); let translatedText = "";
                for (let verse of verses) {
                    if (verse.length > 450) {
                        const lines = verse.split('\n');
                        for (let line of lines) { translatedText += await fetchTranslationChunk(line, targetLang) + "\n"; await new Promise(r => setTimeout(r, 200)); }
                    } else { translatedText += await fetchTranslationChunk(verse, targetLang) + "\n\n"; await new Promise(r => setTimeout(r, 200)); }
                }
                lyricsType = 'plain'; lyricsData = translatedText.split('\n').filter(l => l.trim() !== ''); renderLyrics();
                const badge = document.createElement('div'); badge.className = 'lyrics-translated-badge'; badge.textContent = `Translated to ${currentLyricsLangSpan.textContent}`; lyricsContent.prepend(badge);
            } catch (error) { lyricsContent.innerHTML = `<p style="color:#666; margin-top:50px;">Translation failed.</p>`; }
        }
        async function fetchTranslationChunk(text, targetLang) {
            const url = `https://api.mymemory.translated.net/get?q=${encodeURIComponent(text)}&langpair=autodetect|${targetLang}`;
            const res = await fetch(url), data = await res.json();
            return data.responseStatus === 200 ? data.responseData.translatedText : text;
        }
        async function fetchLyrics(song) {
            if (!song) return;
            lyricsContent.innerHTML = '<div class="spinner"></div>'; lyricsData = []; lyricsType = 'none'; originalPlainLyrics = "";
            try {
                const query = `artist_name=${encodeURIComponent(song.artist)}&track_name=${encodeURIComponent(song.title)}`, url = `https://lrclib.net/api/get?${query}`, response = await fetch(url);
                if (response.ok) {
                    const data = await response.json();
                    if (data) {
                        if (data.syncedLyrics) { lyricsData = parseLRC(data.syncedLyrics); lyricsType = 'synced'; originalPlainLyrics = data.plainLyrics || lyricsData.map(l => l.text).join('\n'); }
                        else if (data.plainLyrics) { originalPlainLyrics = data.plainLyrics; lyricsData = data.plainLyrics.split('\n').filter(line => line.trim() !== ''); lyricsType = 'plain'; }
                        else throw new Error("No lyrics");
                        renderLyrics();
                    } else throw new Error("Empty");
                } else throw new Error("API Error");
            } catch (error) { lyricsContent.innerHTML = `<p style="color:#666; margin-top:50px;">Lyrics not found.</p><div class="lyrics-fallback"><a href="https://www.google.com/search?q=lyrics+${encodeURIComponent(song.artist + ' ' + song.title)}" class="focusable" tabindex="0" target="_blank"><i class="ri-search-line"></i> Search Google</a></div>`; }
        }
        function renderLyrics() {
            lyricsContent.innerHTML = '';
            if (lyricsType === 'synced') { lyricsData.forEach((line, i) => { const p = document.createElement('p'); p.textContent = line.text; p.id = `lyric-${i}`; p.dataset.time = line.time; lyricsContent.appendChild(p); }); }
            else if (lyricsType === 'plain') { lyricsData.forEach((line, i) => { const p = document.createElement('p'); p.textContent = line; p.id = `lyric-${i}`; p.style.opacity = "0.8"; p.style.color = "#ccc"; lyricsContent.appendChild(p); }); }
        }
        function updateLyrics() {
            if (!lyricsModal.classList.contains('open')) return; if (lyricsType !== 'synced' || lyricsData.length === 0) return;
            const currentTime = audio.currentTime; let activeIndex = -1;
            for (let i = 0; i < lyricsData.length; i++) { if (currentTime >= lyricsData[i].time) activeIndex = i; else break; }
            const lines = document.querySelectorAll('.lyrics-content p');
            const currentActive = document.querySelector('.lyrics-content p.active');
            if (currentActive && currentActive.id === `lyric-${activeIndex}`) return;
            lines.forEach(p => p.classList.remove('active'));
            if (activeIndex !== -1) { const activeEl = document.getElementById(`lyric-${activeIndex}`); if (activeEl) { activeEl.classList.add('active'); activeEl.scrollIntoView({ behavior: 'smooth', block: 'center' }); } }
        }
