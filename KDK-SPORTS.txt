        async function fetchHeroBackgrounds() {
            try {
                const url = `https://docs.google.com/spreadsheets/d/${HERO_SHEET_ID}/gviz/tq?tqx=out:json`;
                const res = await fetch(url);
                const text = await res.text();
                const match = text.match(/google\.visualization\.Query\.setResponse\(([\s\S\w]+)\);/);
                
                if (match) {
                    const json = JSON.parse(match[1]);
                    const rows = json.table.rows;
                    if (rows.length > 0) {
                        bgSlides = []; 
                        rows.forEach(row => {
                            if (row.c[0] && row.c[0].v) bgSlides.push(row.c[0].v);
                        });

                        if(bgSlides.length > 0) initHeroSlider();
                    }
                }
            } catch (e) { console.log("Google Sheets Sync Failed."); }
        }

        function initHeroSlider() {
            const dotsContainer = document.getElementById('heroDots');
            dotsContainer.innerHTML = '';
            bgSlides.forEach((_, i) => {
                dotsContainer.innerHTML += `<div class="dot ${i === 0 ? 'active' : ''}"></div>`;
            });
            currentBgIndex = 0;
            updateHeroImage();
            setInterval(() => {
                currentBgIndex++;
                updateHeroImage();
            }, 6000); 
        }

        function updateHeroImage() {
            if (bgSlides.length === 0) return;
            const imgUrl = bgSlides[currentBgIndex % bgSlides.length];
            const heroSec = document.getElementById('hero-section');
            
            heroSec.style.opacity = 0.8;
            setTimeout(() => {
                heroSec.style.backgroundImage = `url('${imgUrl}')`;
                document.querySelectorAll('.dot').forEach((d, i) => {
                    d.className = i === (currentBgIndex % bgSlides.length) ? 'dot active' : 'dot';
                });
                heroSec.style.opacity = 1;
            }, 300);
        }

        function startCountdown(targetDateStr) {
            if(countdownTimer) clearInterval(countdownTimer);
            const targetDate = new Date(targetDateStr).getTime();
            
            countdownTimer = setInterval(() => {
                const now = new Date().getTime();
                const distance = targetDate - now;
                
                if (distance < 0) {
                    clearInterval(countdownTimer);
                    document.getElementById('match-target').innerText = "Match Starting Now!";
                    fetchLiveHubAndTicker(); 
                    return;
                }
                
                const days = Math.floor(distance / (1000 * 60 * 60 * 24));
                const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
                const seconds = Math.floor((distance % (1000 * 60)) / 1000);
                
                let timeStr = "";
                if(days > 0) timeStr += `${days}d `;
                timeStr += `${hours}h ${minutes}m ${seconds}s`;
                
                document.getElementById('match-target').innerText = `Starts in: ${timeStr}`;
            }, 1000);
        }

        async function fetchLiveHubAndTicker() {
            try {
                const currData = await smartFetch('currentMatches', '');
                const scheduleData = await smartFetch('matches', '&offset=0');
                
                const currIpl = currData.data.filter(m => m.series_id === IPL_ID || m.name.toUpperCase().includes("IPL"));
                const schedIpl = scheduleData.data.filter(m => m.series_id === IPL_ID || m.name.toUpperCase().includes("IPL"));
                
                let allMatches = [...currIpl, ...schedIpl];
                const uniqueIds = new Set();
                const combined = [];
                for(let m of allMatches) {
                    if(!uniqueIds.has(m.id)) { uniqueIds.add(m.id); combined.push(m); }
                }
                
                combined.sort((a, b) => new Date(a.dateTimeGMT) - new Date(b.dateTimeGMT));
                
                let liveMatch = null, nextMatch = null, lastMatch = null;
                const now = new Date();

                for (let m of combined) {
                    const matchDate = new Date(m.dateTimeGMT);
                    if (m.matchStarted && !m.matchEnded) { liveMatch = m; } 
                    else if (m.matchEnded || (m.matchStarted && m.matchEnded)) { lastMatch = m; } 
                    else if (!m.matchStarted && matchDate > now && !nextMatch) { nextMatch = m; }
                }

                // 1. UPDATE TICKER
                let tickerText = "<span>🏏 KDK11 API ACTIVE</span> | ";
                if (lastMatch) { tickerText += `⏮️ YESTERDAY'S RESULT: ${lastMatch.name} - ${lastMatch.status} | `; }
                if (liveMatch) { tickerText += `🔴 LIVE NOW: ${getTeamCode(liveMatch.teams[0])} vs ${getTeamCode(liveMatch.teams[1])} | `; } 
                else if (nextMatch) {
                    const nDate = new Date(nextMatch.dateTimeGMT);
                    tickerText += `⏱️ NEXT UP: ${nextMatch.name} starting at ${nDate.toLocaleTimeString('en-US', {hour: '2-digit', minute:'2-digit'})} | `;
                }
                document.getElementById('newsTicker').innerHTML = tickerText;

                // 2. SMART HERO BADGE LOGIC
                const b = document.getElementById('match-badge');
                if(countdownTimer) clearInterval(countdownTimer); 

                if (liveMatch) {
                    activeMatchId = liveMatch.id;
                    b.innerText = "● LIVE NOW";
                    b.className = "badge badge-live";
                    document.getElementById('hero-title').innerText = `${getTeamCode(liveMatch.teams[0])} VS ${getTeamCode(liveMatch.teams[1])}`;
                    
                    const scoreData = await smartFetch('match_scorecard', `&id=${liveMatch.id}`);
                    const m = scoreData.data;
                    let scoreStr = "Fetching Live Data...";
                    if(m.score && m.score.length > 0) {
                        scoreStr = `Live Score: ${m.score[0].r}/${m.score[0].w} (${m.score[0].o})`;
                    }
                    document.getElementById('match-target').innerText = scoreStr;
                    updateScorecardUI(m);
                } 
                else if (nextMatch) {
                    b.innerText = "UPCOMING";
                    b.className = "badge badge-upcoming";
                    document.getElementById('hero-title').innerText = `${getTeamCode(nextMatch.teams[0])} VS ${getTeamCode(nextMatch.teams[1])}`;
                    startCountdown(nextMatch.dateTimeGMT);
                } 
                else if (lastMatch) {
                    activeMatchId = lastMatch.id;
                    b.innerText = "HIGHLIGHTS";
                    b.className = "badge badge-highlights";
                    document.getElementById('hero-title').innerText = `${getTeamCode(lastMatch.teams[0])} VS ${getTeamCode(lastMatch.teams[1])}`;
                    document.getElementById('match-target').innerText = lastMatch.status;
                    
                    const scoreData = await smartFetch('match_scorecard', `&id=${lastMatch.id}`);
                    updateScorecardUI(scoreData.data);
                } else {
                    b.innerText = "STANDBY";
                    b.className = "badge badge-standby";
                    document.getElementById('match-target').innerText = "Awaiting Tournament Data...";
                }
                
            } catch (e) { console.log("Live Hub Error"); }
        }

        // ==========================================
        // SCORECARD UI ENGINE
        // ==========================================
        function updateScorecardUI(m) {
            if (!m) return;
            document.getElementById('live-comm').innerText = m.status;
            
            // Set Logos
            document.getElementById('t1-logo').src = getTeamLogo(m.teams[0]);
            document.getElementById('t2-logo').src = getTeamLogo(m.teams[1]);

            // Set Names
            document.getElementById('t1-name').innerText = getTeamCode(m.teams[0]);
            document.getElementById('t2-name').innerText = getTeamCode(m.teams[1]);

            if(m.score && m.score[0]) {
                document.getElementById('t1-score').innerText = `${m.score[0].r}/${m.score[0].w}`;
                document.getElementById('t1-ov').innerText = `(${m.score[0].o})`;
            }
            if(m.score && m.score[1]) {
                document.getElementById('t2-score').innerText = `${m.score[1].r}/${m.score[1].w}`;
                document.getElementById('t2-ov').innerText = `(${m.score[1].o})`;
            }

            if (m.status !== prevStatus) {
                let s = m.status.toLowerCase();
                if (s.includes("six")) triggerAnim('SIX', 'var(--accent)');
                else if (s.includes("four")) triggerAnim('FOUR', 'var(--blue)');
                else if (s.includes("wicket") || s.includes("out")) triggerAnim('WICKET', 'var(--live)');
                if (s.includes("won")) confetti();
                prevStatus = m.status;
            }
        }

        // TV REMOTE NAVIGATION LOGIC
        // ==========================================
        let focusIndex = 0;
        function updateFocus() {
            document.querySelectorAll('.focussable').forEach(el => el.classList.remove('active-focus'));
            const isModalOpen = document.getElementById('langModal').style.display === 'flex';
            const isGameHubOpen = document.getElementById('gamePlayer').style.display === 'block';

            let items;
            if (isModalOpen) {
                items = document.querySelectorAll('#langModal .focussable');
            } else if (isGameHubOpen) {
                items = document.querySelectorAll('#gamePlayer .focussable');
            } else {
                items = document.querySelectorAll('.active-view .focussable');
            }
            
            const visibleItems = Array.from(items).filter(el => el.offsetWidth > 0 && el.offsetHeight > 0);
            
            if(visibleItems.length > 0) {
                if(focusIndex >= visibleItems.length) focusIndex = 0;
                if(focusIndex < 0) focusIndex = visibleItems.length - 1;
                visibleItems[focusIndex].focus();
                visibleItems[focusIndex].classList.add('active-focus'); 
            }
        }
        
        document.addEventListener('keydown', (e) => {
            if (e.key === 'ArrowDown' || e.key === 'ArrowRight') { e.preventDefault(); focusIndex++; updateFocus(); } 
            else if (e.key === 'ArrowUp' || e.key === 'ArrowLeft') { e.preventDefault(); focusIndex--; updateFocus(); } 
            else if (e.key === 'Enter') { e.preventDefault(); document.activeElement.click(); }
        });
